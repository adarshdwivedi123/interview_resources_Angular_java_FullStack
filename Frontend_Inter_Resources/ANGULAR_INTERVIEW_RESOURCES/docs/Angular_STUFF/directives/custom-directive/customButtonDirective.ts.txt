import {Directive ,ElementRef,HostListener,Input} from '@angular/core';

@Directive({
  selector:'[appCustomButton]'

})

export class AppCustomButtonDirective {
   @Input() customColor :string =' ';
    constructor(private elementRef: ElementRef){}


@HostListener('mouseenter') omMouseEnter(){
     this.applyCustomStyle(this.customColor);
}

@HostListener('mouseleave') omMouseLeave(){
     this.applyCustomStyle(null);
}

private applyCustomStyle(color:any){
this.elementRef.nativeElement.style.backgroundColor=color;
}



}

✅ 2. Disable Button Directive
👉 Question

Disable button after one click

💻 Solution
@Directive({
  selector: '[appDisableAfterClick]'
})
export class DisableAfterClickDirective {

  constructor(private el: ElementRef) {}

  @HostListener('click')
  onClick() {
    this.el.nativeElement.disabled = true;
  }
}
📌 Use
<button appDisableAfterClick>Submit</button>
💡 Explanation

✔ Prevents multiple clicks (real use case)

import { Directive, ElementRef, AfterViewInit } from '@angular/core';

@Directive({
  selector: '[appAutoFocus]'
})
export class AutoFocusDirective implements AfterViewInit {

  constructor(private el: ElementRef) {}

  ngAfterViewInit() {
    this.el.nativeElement.focus();
  }
}

🔴 9. Custom *ngIf
@Directive({
  selector: '[appIf]'
})
export class AppIfDirective {

  constructor(
    private templateRef: TemplateRef<any>,
    private viewContainer: ViewContainerRef
  ) {}

  @Input() set appIf(condition: boolean) {
    this.viewContainer.clear();

    if (condition) {
      this.viewContainer.createEmbeddedView(this.templateRef);
    }
  }
}

✅ Role Directive (Final Version)
import { Directive, Input, TemplateRef, ViewContainerRef, OnInit } from '@angular/core';

@Directive({
  selector: '[appRole]'
})
export class RoleDirective implements OnInit {

  @Input() appRole!: string | string[]; // support single or multiple roles

  private userRole: string = '';

  constructor(
    private templateRef: TemplateRef<any>,
    private viewContainer: ViewContainerRef
  ) {}

  ngOnInit() {
    // 🔥 In real project → get from AuthService / API
    this.userRole = 'admin'; // example

    this.updateView();
  }

  private updateView() {
    this.viewContainer.clear();

    if (this.checkAccess()) {
      this.viewContainer.createEmbeddedView(this.templateRef);
    }
  }

  private checkAccess(): boolean {
    if (Array.isArray(this.appRole)) {
      return this.appRole.includes(this.userRole);
    }
    return this.userRole === this.appRole;
  }
}
